#include <stdio.h>
#include <math.h>

int main() {
int n;
float life = 10.0;
scanf(“%d”, &n);
float total_exp = 0.0;
int valid_operations = n; 

for (int i = 0; i < n; i++) {
    float x, a;
    scanf("%f %f", &x, &a);
    if (life <= 0) {
        valid_operations = i;
        break;
    }
    life -= x;
    if (life > 10.0) {
        life = 10.0;
    }
    if (life < 0.0) {
        life = 0.0;
    }
            if (life > 0 && a > 0) {
        total_exp += a;
    }
}

int level = 0;
double remain = total_exp;

while (remain>= pow(2, level)) {
    remain -= pow(2, level);
    level++;
}
printf("%d %.0lf\n", level, remain);

return 0;

}
