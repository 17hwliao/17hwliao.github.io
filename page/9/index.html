实现数字反转
#include <stdio.h>

int add2(int a, int c) {
if (a == 0) {
return c;
}
return add2(a / 10, c * 10 + a % 10);
}

int main() {
int x = 0;
int y = 0;
scanf(“%d”,&x);
if (x < 0) {
x = -x;
y = add2(x, 0);
printf(“-%d”, y);
} else {
y = add2(x, 0);
printf(“%d”, y);
}
return 0;
}

game.c:
#include “game.h”

void in_board(char board[ROWS][COLS], int rows, int cols, char set) {
int i = 0;
int j = 0;
for (i = 0; i < rows; i++) {
for (j = 0; j < cols; j++) {
board[i][j] = set;
}
}
}

void paly_board(char board[ROWS][COLS], int row, int col) {
int i = 0;
int j = 0;
printf(“—————-扫雷游戏——————-\n”);
for (j = 0; j <= col; j++) {
printf(“%d “, j);
}
printf(“\n”);
for (i = 1; i <= row; i++) {
printf(“%d “, i);
for (j = 1; j <= col; j++) {
printf(“%c “, board[i][j]);
}
printf(“\n”);
}
printf(“—————-扫雷游戏——————-\n”);
}

void setmine(char board[ROWS][COLS], int row, int col) {
int count = MINE_COUNT;
while (count) {
int x = rand() % row + 1;
int y = rand() % col + 1;
if (board[x][y] == ‘0’) {
board[x][y] = ‘1’;
count–;
}
}
}

int get_mine_count(char board[ROWS][COLS], int x, int y) {
return (board[x - 1][y] +
board[x - 1][y - 1] +
board[x - 1][y + 1] +
board[x][y - 1] +
board[x][y + 1] +
board[x + 1][y - 1] +
board[x + 1][y + 1] +
board[x + 1][y] - 8 * ‘0’);
}

// 扩展空白区域的递归函数
void expand(char mine[ROWS][COLS], char show[ROWS][COLS], int x, int y, int* win) {
// 检查坐标是否越界
if (x < 1 || x > ROW || y < 1 || y > COL) {
return;
}

// 如果该位置已经展开或者是地雷，则返回
if (show[x][y] != '*' || mine[x][y] == '1') {
	return;
}

// 计算周围地雷数量
int count = get_mine_count(mine, x, y);
show[x][y] = count + '0';
(*win)++;

// 如果周围没有地雷，则递归展开周围的8个格子
if (count == 0) {
	expand(mine, show, x - 1, y - 1, win);
	expand(mine, show, x - 1, y, win);
	expand(mine, show, x - 1, y + 1, win);
	expand(mine, show, x, y - 1, win);
	expand(mine, show, x, y + 1, win);
	expand(mine, show, x + 1, y - 1, win);
	expand(mine, show, x + 1, y, win);
	expand(mine, show, x + 1, y + 1, win);
}

}

void FindMine(char mine[ROWS][COLS], char show[ROWS][COLS], int row, int col) {
int x = 0;
int y = 0;
int win = 0;

while (win < row * col - MINE_COUNT) {
	printf("牢底祝你这次不会被炸死，坐标>");
	scanf_s("%d %d", &x, &y);
	if (x >= 1 && x <= row && y >= 1 && y <= col) {
		if (show[x][y] != '*') {
			printf("会玩不这已经有了\n");
		}
		else {
			if (mine[x][y] == '1') {
				printf("很遗憾，您踩到地雷了！游戏结束\n");
				paly_board(mine, ROW, COL);
				break;
			}
			else {
				// 使用扩展函数而不是直接显示数字
				expand(mine, show, x, y, &win);
				paly_board(show, ROW, COL);
			}
		}
	}
	else {
		printf("牢底会不会玩，重新输入！\n");
	}

	if (win == row * col - MINE_COUNT) {
		printf("牢底你会玩，赢了\n");
		paly_board(show, ROW, COL);
	}
}

}

int ping(char board[ROW][COL], int row, int col) {
int i = 0;
int j = 0;
for (i = 0; i < col; i++) {
for (j = 0; j < col; j++) {
if (board[i][j] == ‘ ‘) {
return 0;
}
}
}
return 1;
}
char win(char board[ROW][COL], int row, int col) {
int i = 0; //行
for (i = 0; i < row; i++) {
if (board[i][0] == board[i][1] && board[i][2] && board[i][1] != ‘ ‘) {
return board[i][1];
}
}
int j = 0; //列
for (j = 0; j < row; j++) {
if (board[0][j] == board[1][j] && board[2][j] && board[1][j] != ‘ ‘) {
return board[1][j];
}
}
if(board[0][2] == board[1][1] && board[2][0] == board[1][1] && board[1][1] != ‘ ‘){
return board[1][1]; } //对角线

if (ping(board, row, col)) {
return’Q’;
}
return’C’;
}

out_board(board, ROW, COL); //耍时候打印的棋盘
