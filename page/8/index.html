计算素数和素数和：
#include <stdio.h>
#include <math.h>

int prime(int p) {
if (p < 2) return 0;
if (p == 2) return 1;
if (p % 2 == 0) return 0; 

int limit = (int)sqrt(p);
for (int i = 3; i <= limit; i += 2) {
    if (p % i == 0) {
        return 0;
    }
}
return 1;

}

int PrimeSum(int m, int n) {
int sum = 0;
for (int i = m; i <= n; i++) {
if (prime(i)) {
sum += i;
}
}
return sum;
}

//冒泡排序
void func1(int arr[], int n) {
int i, j;
for (i = 0; i < n - 1; i++) {
for (j = 0; j < n - 1 - i; j++) {
if (arr[j] > arr[j + 1]) {
int num = arr[j];
arr[j] = arr[j+1];
arr[j + 1] = num;
}
}
}
}

//二分查找
int func2(int n, int arr[],int x) {
int left = 0;
int right = n - 1;
while (left <= right) {
int mid = left +(right - left) / 2;
if (arr[mid] == x) {
return mid;
}
else if (arr[mid] < x) {
left = mid + 1;
}
else right = mid - 1;
}
return -1;
}

//希尔排序
void func3(int arr[], int n) {
int i, j, num,gap;
for (gap = n / 2; gap > 0; gap /= 2) {
for (i = gap; i<n; i++) {
num = arr[i];
for (j = i; j >= gap && num < arr[j - gap]; j -= gap) {
arr[j] = arr[j - gap];
}
arr[j] = num;
}
}
}

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define MAX_NAME_LENGTH 50
#define MAX_NAMES 100000

typedef struct {
char name[MAX_NAME_LENGTH];
int count;
} Child;

Child children[MAX_NAMES];
int child_count = 0;
int find_or_add_child(const char *name) {
for (int i = 0; i < child_count; i++) {
if (strcmp(children[i].name, name) == 0) {
return i;
}
}
strcpy(children[child_count].name, name);
children[child_count].count = 0;
return child_count++;
}

int main() {
int n;
scanf(“%d”, &n);

int total = 0;
int ans = 0;

for (int i = 0; i < n; i++) {
    char name[MAX_NAME_LENGTH];
    scanf("%s", name);
    
    int idx = find_or_add_child(name);
    
    if (children[idx].count > total - children[idx].count) {
        ans++;
    }
    
    children[idx].count++;
    total++;
}

printf("%d\n", ans);
return 0;

}

include <stdio.h>
#include <stdbool.h>

int main() {
int n, m, horse_x, horse_y;
scanf(“%d %d %d %d”, &n, &m, &horse_x, &horse_y);

// 马的控制点偏移量
int dx[] = {1, 1, -1, -1, 2, 2, -2, -2, 0};
int dy[] = {2, -2, 2, -2, 1, -1, 1, -1, 0};

// 创建障碍物网格
bool obstacle[n + 1][m + 1];
for (int i = 0; i <= n; i++) {
    for (int j = 0; j <= m; j++) {
        obstacle[i][j] = false;
    }
}

// 标记马的控制点为障碍物
for (int i = 0; i < 9; i++) {
    int nx = horse_x + dx[i];
    int ny = horse_y + dy[i];
    if (nx >= 0 && nx <= n && ny >= 0 && ny <= m) {
        obstacle[nx][ny] = true;
    }
}

// 如果起点是障碍物，直接输出0
if (obstacle[0][0]) {
    printf("0\n");
    return 0;
}

// 初始化DP数组
long long dp[n + 1][m + 1];
for (int i = 0; i <= n; i++) {
    for (int j = 0; j <= m; j++) {
        dp[i][j] = 0;
    }
}
dp[0][0] = 1;

// 动态规划计算路径数
for (int i = 0; i <= n; i++) {
    for (int j = 0; j <= m; j++) {
        if (obstacle[i][j]) {
            continue;
        }
        if (i > 0 && !obstacle[i - 1][j]) {
            dp[i][j] += dp[i - 1][j];
        }
        if (j > 0 && !obstacle[i][j - 1]) {
            dp[i][j] += dp[i][j - 1];
        }
    }
}

printf("%lld\n", dp[n][m]);
return 0;

}

void setmine(char board[ROWS][COLS], int row, int col) {
int count = MINE_COUNT;
while (count) {
int x = rand() % row + 1;
int y = rand() % col + 1;
if (board[x][y] == ‘0’) {
board[x][y] = ‘1’;
count–;
}
}
}

int get_mine_count(char board[ROWS][COLS], int x, int y) {
return (board[x - 1][y] +
board[x - 1][y - 1] +
board[x - 1][y + 1] +
board[x][y - 1] +
board[x][y + 1] +
board[x + 1][y - 1] +
board[x + 1][y + 1] +
board[x + 1][y] - 8 * ‘0’);
}

// 扩展空白区域的递归函数
void expand(char mine[ROWS][COLS], char show[ROWS][COLS], int x, int y, int* win) {
// 检查坐标是否越界
if (x < 1 || x > ROW || y < 1 || y > COL) {
return;
}

// 如果该位置已经展开或者是地雷，则返回
if (show[x][y] != '*' || mine[x][y] == '1') {
	return;
}

// 计算周围地雷数量
int count = get_mine_count(mine, x, y);
show[x][y] = count + '0';
(*win)++;

// 如果周围没有地雷，则递归展开周围的8个格子
if (count == 0) {
	expand(mine, show, x - 1, y - 1, win);
	expand(mine, show, x - 1, y, win);
	expand(mine, show, x - 1, y + 1, win);
	expand(mine, show, x, y - 1, win);
	expand(mine, show, x, y + 1, win);
	expand(mine, show, x + 1, y - 1, win);
	expand(mine, show, x + 1, y, win);
	expand(mine, show, x + 1, y + 1, win);
}

}

void FindMine(char mine[ROWS][COLS], char show[ROWS][COLS], int row, int col) {
int x = 0;
int y = 0;
int win = 0;

while (win < row * col - MINE_COUNT) {
	printf("牢底祝你这次不会被炸死，坐标>");
	scanf_s("%d %d", &x, &y);
	if (x >= 1 && x <= row && y >= 1 && y <= col) {
		if (show[x][y] != '*') {
			printf("会玩不这已经有了\n");
		}
		else {
			if (mine[x][y] == '1') {
				printf("很遗憾，您踩到地雷了！游戏结束\n");
				paly_board(mine, ROW, COL);
				break;
			}
			else {
				// 使用扩展函数而不是直接显示数字
				expand(mine, show, x, y, &win);
				paly_board(show, ROW, COL);
			}
		}
	}
	else {
		printf("牢底会不会玩，重新输入！\n");
	}

	if (win == row * col - MINE_COUNT) {
		printf("牢底你会玩，赢了\n");
		paly_board(show, ROW, COL);
	}
}

}
