#include <stdio.h>
#include <math.h>

int narcissistic(int number) {
int a = number;
int n = 0;
int sum = 0;

int temp = number;
while (temp != 0) {
    n++;
    temp /= 10;
}

temp = number;
while (temp != 0) {
    int digit = temp % 10;
    sum += pow(digit, n);
    temp /= 10;
}

if (sum == a) {
    return 1;
} else {
    return 0;
}

}

void PrintN(int m, int n) {
for (int i = m + 1; i < n; i++) {
if (narcissistic(i)) {
printf(“%d\n”, i);
}
}
}

int CountDigit(int number, int digit) {
if (number == 0 && digit == 0){
return 1;
}
int count = 0;
long long n = number;
if (n < 0) {
n = -n;
}
while (n > 0) {
int d = n % 10;
if (d == digit) {
count++;
}
n /= 10;
}
return count;
}

计算素数和素数和：
#include <stdio.h>
#include <math.h>

int prime(int p) {
if (p < 2) return 0;
if (p == 2) return 1;
if (p % 2 == 0) return 0; 

int limit = (int)sqrt(p);
for (int i = 3; i <= limit; i += 2) {
    if (p % i == 0) {
        return 0;
    }
}
return 1;

}

int PrimeSum(int m, int n) {
int sum = 0;
for (int i = m; i <= n; i++) {
if (prime(i)) {
sum += i;
}
}
return sum;
}

//冒泡排序
void func1(int arr[], int n) {
int i, j;
for (i = 0; i < n - 1; i++) {
for (j = 0; j < n - 1 - i; j++) {
if (arr[j] > arr[j + 1]) {
int num = arr[j];
arr[j] = arr[j+1];
arr[j + 1] = num;
}
}
}
}

//二分查找
int func2(int n, int arr[],int x) {
int left = 0;
int right = n - 1;
while (left <= right) {
int mid = left +(right - left) / 2;
if (arr[mid] == x) {
return mid;
}
else if (arr[mid] < x) {
left = mid + 1;
}
else right = mid - 1;
}
return -1;
}

//希尔排序
void func3(int arr[], int n) {
int i, j, num,gap;
for (gap = n / 2; gap > 0; gap /= 2) {
for (i = gap; i<n; i++) {
num = arr[i];
for (j = i; j >= gap && num < arr[j - gap]; j -= gap) {
arr[j] = arr[j - gap];
}
arr[j] = num;
}
}
}

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define MAX_NAME_LENGTH 50
#define MAX_NAMES 100000

typedef struct {
char name[MAX_NAME_LENGTH];
int count;
} Child;

Child children[MAX_NAMES];
int child_count = 0;
int find_or_add_child(const char *name) {
for (int i = 0; i < child_count; i++) {
if (strcmp(children[i].name, name) == 0) {
return i;
}
}
strcpy(children[child_count].name, name);
children[child_count].count = 0;
return child_count++;
}

int main() {
int n;
scanf(“%d”, &n);

int total = 0;
int ans = 0;

for (int i = 0; i < n; i++) {
    char name[MAX_NAME_LENGTH];
    scanf("%s", name);
    
    int idx = find_or_add_child(name);
    
    if (children[idx].count > total - children[idx].count) {
        ans++;
    }
    
    children[idx].count++;
    total++;
}

printf("%d\n", ans);
return 0;

}

include <stdio.h>
#include <stdbool.h>

int main() {
int n, m, horse_x, horse_y;
scanf(“%d %d %d %d”, &n, &m, &horse_x, &horse_y);

// 马的控制点偏移量
int dx[] = {1, 1, -1, -1, 2, 2, -2, -2, 0};
int dy[] = {2, -2, 2, -2, 1, -1, 1, -1, 0};

// 创建障碍物网格
bool obstacle[n + 1][m + 1];
for (int i = 0; i <= n; i++) {
    for (int j = 0; j <= m; j++) {
        obstacle[i][j] = false;
    }
}

// 标记马的控制点为障碍物
for (int i = 0; i < 9; i++) {
    int nx = horse_x + dx[i];
    int ny = horse_y + dy[i];
    if (nx >= 0 && nx <= n && ny >= 0 && ny <= m) {
        obstacle[nx][ny] = true;
    }
}

// 如果起点是障碍物，直接输出0
if (obstacle[0][0]) {
    printf("0\n");
    return 0;
}

// 初始化DP数组
long long dp[n + 1][m + 1];
for (int i = 0; i <= n; i++) {
    for (int j = 0; j <= m; j++) {
        dp[i][j] = 0;
    }
}
dp[0][0] = 1;

// 动态规划计算路径数
for (int i = 0; i <= n; i++) {
    for (int j = 0; j <= m; j++) {
        if (obstacle[i][j]) {
            continue;
        }
        if (i > 0 && !obstacle[i - 1][j]) {
            dp[i][j] += dp[i - 1][j];
        }
        if (j > 0 && !obstacle[i][j - 1]) {
            dp[i][j] += dp[i][j - 1];
        }
    }
}

printf("%lld\n", dp[n][m]);
return 0;

}
