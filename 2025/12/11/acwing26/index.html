ph[] : 实现找出第K个插入数的下标( 指针有单向性 ,索引出需要使用另外数组)
hp[] : 实现找出下标为K的数的插入序号
hp[i] = j ; ph[j] = i ;

//实现交换下标和插入顺序的交换
void heap_swap(int a, int b)
{
swap(ph[hp[a]],ph[hp[b]]);
swap(hp[a], hp[b]);
swap(h[a], h[b]);
}

void down(int u)
{
int t = u;
if (u * 2 <= cnt && h[u * 2] < h[t]) t = u * 2;
if (u * 2 + 1 <= cnt && h[u * 2 + 1] < h[t]) t = u * 2 + 1;
if (u != t)
{
heap_swap(u, t);
down(t);
}
}

//实现上浮操作
void up(int u)
{
while (u / 2 && h[u] < h[u / 2])
{
heap_swap(u, u / 2);
u >>= 1;
}
}

using namespace std;
const int N = 1e5 + 3;
int h[N], e[N], ne[N], idx;

//实现哈希找数
int hash_(int x) {
return (x % N + N) % N;
}

//实现存储数值
void insert(int x) {
int k = hash_(x);
e[idx] = x;
ne[idx] = h[k];
h[k] = idx++;
}

//实现查找功能
bool find(int x) {
int k = hash_(x);
for (int i = h[k]; i != -1; i = ne[i]) {
if (e[i] == x) return true;
}
return false;
}

//实现删除功能
void remove(int x) {
int k = hash_(x);
int head = -1;
for (int i = h[k]; i != -1; i = ne[i]) {
if (e[i] == x) {
if (head == -1) { // i是头节点 ;
h[k] = ne[i]; //头指针直接向下一个☞
}
else { //i是中间节点或尾节点情况
ne[head] = ne[i]; //跳过 i , 连接 i 的下一个
}
return;
}
head = i; //更新head为当前节点,成为下一个的前一个节点
}
}
