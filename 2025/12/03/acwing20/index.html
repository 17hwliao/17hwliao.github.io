#include
#include
using namespace std;

// 双向链表节点结构
typedef struct node {
int data;
struct node* prev; // 前驱指针（应该是prev，不是l）
struct node* next; // 后继指针（应该是next）
} node;

// 双向链表结构
typedef struct list {
int size;
node* head;
node* end;
} lian;

// 创建新节点
node* create_node(int x) {
node* newnode = (node*)malloc(sizeof(node));
if (newnode == NULL) {
return NULL;
}
newnode->data = x;
newnode->prev = NULL;
newnode->next = NULL;
return newnode;
}

// 初始化链表
void init(lian* arr) {
arr->head = NULL;
arr->end = NULL;
arr->size = 0;
}

// 头插法
void add_to_head(int x, lian* arr) {
node* newnode = create_node(x);
if (newnode == NULL) return;

if (arr->size == 0) {
    arr->head = newnode;
    arr->end = newnode;
}
else {
    newnode->next = arr->head;
    arr->head->prev = newnode;
    arr->head = newnode;
}
arr->size++;

}

// 尾插法
void add_to_tail(int x, lian* arr) {
node* newnode = create_node(x);
if (newnode == NULL) return;

if (arr->size == 0) {
    arr->head = newnode;
    arr->end = newnode;
}
else {
    newnode->prev = arr->end;
    arr->end->next = newnode;
    arr->end = newnode;
}
arr->size++;

}

// 在指定位置插入（位置从0开始）
void insert_at(int x, int pos, lian* arr) {
if (pos < 0 || pos > arr->size) {
cout << "位置无效！" << endl;
return;
}

if (pos == 0) {
    add_to_head(x, arr);
    return;
}

if (pos == arr->size) {
    add_to_tail(x, arr);
    return;
}

node* newnode = create_node(x);
if (newnode == NULL) return;

// 找到插入位置的前一个节点
node* current;
if (pos <= arr->size / 2) {
    // 从前向后找
    current = arr->head;
    for (int i = 0; i < pos - 1; i++) {
        current = current->next;
    }
}
else {
    // 从后向前找
    current = arr->end;
    for (int i = arr->size - 1; i > pos - 1; i--) {
        current = current->prev;
    }
}

// 插入新节点
newnode->next = current->next;
newnode->prev = current;
current->next->prev = newnode;
current->next = newnode;

arr->size++;

}

// 删除头节点
void delete_head(lian* arr) {
if (arr->size == 0) {
cout << "链表为空！" << endl;
return;
}

node* temp = arr->head;

if (arr->size == 1) {
    arr->head = NULL;
    arr->end = NULL;
}
else {
    arr->head = arr->head->next;
    arr->head->prev = NULL;
}

free(temp);
arr->size--;

}

// 删除尾节点
void delete_tail(lian* arr) {
if (arr->size == 0) {
cout << "链表为空！" << endl;
return;
}

node* temp = arr->end;

if (arr->size == 1) {
    arr->head = NULL;
    arr->end = NULL;
}
else {
    arr->end = arr->end->prev;
    arr->end->next = NULL;
}

free(temp);
arr->size--;

}

// 删除指定值的节点（删除第一个匹配的）
void delete_value(int x, lian* arr) {
if (arr->size == 0) {
cout << "链表为空！" << endl;
return;
}

node* current = arr->head;
while (current != NULL) {
    if (current->data == x) {
        if (current == arr->head) {
            delete_head(arr);
        }
        else if (current == arr->end) {
            delete_tail(arr);
        }
        else {
            current->prev->next = current->next;
            current->next->prev = current->prev;
            free(current);
            arr->size--;
        }
        cout << "删除成功！" << endl;
        return;
    }
    current = current->next;
}

    current = arr->end;

// 释放链表内存
void free_list(lian* arr) {
node* current = arr->head;
while (current != NULL) {
node* temp = current;
current = current->next;
free(temp);
}
arr->head = NULL;
arr->end = NULL;
arr->size = 0;
}
