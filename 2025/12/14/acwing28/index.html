#include
using namespace std;
const int N = 1e3 + 17;
int n;
int path[N];
bool std[N];

void dfs(int x) {
if (x == n) {
for (int i = 0; i < n; i++) cout << path[i] << ' ';
cout << '\n';
return;
}
for (int i = 1; i <= n; i++) {
if (!std[i]) {
path[x] = i;
std[i] = true; 对放入数据标记
dfs(x + 1); 实现找全
std[i] = false;
}
}
}

int main(){
cin >> n;
dfs(0);
return 0;
}
//DFS算法: //实现找路径总数 ( 关键就是回溯之前要恢复原状 )
int n;
int arr[N];
void dfs(int x, int pos) { //pos : 表示二进制位
if (x == n) { //递归返回n 个数时;
for (int i = 0; i < n; i++) cout << arr[i];
cout << '\n';
return;
}
for (int i = 0; i < n; i++) {
if (!(pos >> i & 1)) { 通过判断该二进制位上是否为0, 判断是否被使用
arr[x] = i + 1; 下标从0开始用+ 1; //在arr数组中下标为 x 的位置存入当前的未标记数
dfs(x + 1, pos + (1 << i));
}
}
}
int main() {
cin >> n;
dfs(0, 0);
return 0;
}

#include
using namespace std;
const int N = 10;
int n, ans;
int pos[N], c[N], p[N *2 ], q[N * 2 ];

void println() {
if (ans <= 3) { //题目要求输出前3组解
for (int i = 1; i <= n; i++) cout << pos[i]<< ' ';
cout << endl;
}
return;
}

void dfs(int i) {
if (i > n) {
ans++;
println();
return;
}
for (int j = 1; j <= n; j++) {
if (c[j] || p[i + j] || q[i - j + n]) continue; //遍历 , 判断 , 传值 ,标记, 向下递归 , 恢复现场 ( 难点: 标记位置 , 传值内容 )
pos[i] = j;
c[j] = p[i + j] = q[i - j + n] = 1; //对位置标记
dfs(i + 1);
c[j] = p[i + j] = q[i - j + n] = 0; //解除标记
}
}
int main() {
cin >> n;
dfs(1);
cout << ans << endl;
return 0;
}
