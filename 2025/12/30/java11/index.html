private  static void Bubble_Sort(int[] arr){
    for(int i= 0  ; i < arr.length -1 ; i ++){
        boolean flag = false;
        for(int j = 0 ; j < arr.length  -1 - i ; j++){
            if(arr[j] > arr[j+1]){
                int temp = arr[j];
                arr[j] = arr[j+1];
                arr[j+1] = temp;
                flag = true ;
            }
        }
        if(!flag) break ;
    }
}

private static int Bin_find( int[]arr , int x ){    //实现最小化查找( >= ) r 返回第一个满足条件的数
    int l = -1 ;                                    //实现最大化查找( <= ) l 返回最后一个满足条件的数
    int r = arr.length  ;
    while(l +  1 < r ){
        int mid = (l + r ) >> 1 ;
        if(arr[mid] <= x) l = mid ;
        else r = mid ;
    }
    return l ;
}

private static int Fibo(int n){
    if( n <= 2 ) return 1 ;
    int[] dp = new int[n+1] ;
    dp[1] = dp[2] = 1 ;
    for(int i = 3 ; i <= n ; i++){
        dp[i] = dp[i -1 ] + dp[i -2] ;
    }
    return dp[n] ;
}
