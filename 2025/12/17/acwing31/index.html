//增加dp动态规划,实现找出所有路径
f[1]=x1, f[2]=x2, f[3]=x3, f[4]=x4, f[5]= x5, f[6]=x6
通过叠加路径数 , 实现找到所有的路径数
const int N = 5e5 + 17;
const int mod = 80112002;
int n, m;
int h[N], e[N], ne[N], idx;
int q[N], d[N] , f[N ] , c[N];

void add(int a, int b) {
e[idx] = b;
ne[idx] = h[a];
h[a] = idx++;
}

void topsort() {
int hh = 0, tt = -1;
for (int i = 1; i <= n; i++) {
if (!d[i]) {
q[++tt] = i;
f[i] = 1;
}
}
while (hh <= tt) {
int t = q[hh++];
for (int i = h[t]; i != -1; i = ne[i]) {
int j = e[i];
f[j] = (f[j] + f[t]) % mod;
d[j]--;
if (!d[j]) q[++tt] = j;
}
}
int ans = 0;
for (int i = 0; i <= n; i++) {
if (!c[i]) {
ans = (ans + f[i]) % mod;
}
}
cout << ans << endl;
}

//寻找最长路径 : //增加数组w 来记录权重 ;
const int N = 1e5 + 17;
const int INF = -0x3f3f3f3f;
int n , m ;
int e[N], ne[N], idx , h[N] , w[N];
int p[N], d[N], f[N];

void add(int a, int b , int c ) {
e[idx] = b;
w[idx] = c;
ne[idx] = h[a];
h[a] = idx++;
}

bool topsort() {
int hh = 0, tt = -1;
for (int i = 1; i <= n; i++) {
f[i] = INF;
}
f[1] = 0;
for (int i = 1; i <= n; i++) {
if (!d[i]) {
p[++tt] = i;
}
}
while (hh <= tt) {
int t = p[hh++];
for (int i = h[t]; i != -1; i = ne[i]) {
int j = e[i];
if (f[t] > INF) {
f[j] = max(f[j], f[t] + w[i]);
}
d[j]--;
if (!d[j]) p[++tt] = j;
}
}
return tt == n - 1;
}
