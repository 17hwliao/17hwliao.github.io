特殊技巧：后序的最后一个是根节点
//通过中序和后序找出先序
map<char, int> arr;
string a, b;

void dfs(int l1 , int r1 , int l2 , int r2) {
if (l1 > r1 ) return;
char jie = b[r2];
cout << jie;
int pos = arr[jie];
int l_size = pos - l1; //找出中序分界点
dfs(l1, pos - 1, l2, l2 + l_size - 1); //在中序和后序中分别找出左子树
dfs(pos + 1, r1, l2 + l_size, r2 - 1); //在中序和后序中分别找出右子树
}

int main() {
cin >> a >> b;
int n = a.length();
for (int i = 0; i < a.length(); i++) {
arr[a[i]] = i;
}
dfs(0, n-1, 0, n-1);
return 0;
}
