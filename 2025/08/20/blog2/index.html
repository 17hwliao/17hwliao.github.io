实现扫雷游戏：
game.c:
#include “game.h”

void in_board(char board[ROWS][COLS], int rows, int cols, char set) {
int i = 0;
int j = 0;
for (i = 0; i < rows; i++) {
for (j = 0; j < cols; j++) {
board[i][j] = set;
}
}
}

void paly_board(char board[ROWS][COLS], int row, int col) {
int i = 0;
int j = 0;
printf(“—————-扫雷游戏——————-\n”);
for (j = 0; j <= col; j++) {
printf(“%d “, j);
}
printf(“\n”);
for (i = 1; i <= row; i++) {
printf(“%d “, i);
for (j = 1; j <= col; j++) {
printf(“%c “, board[i][j]);
}
printf(“\n”);
}
printf(“—————-扫雷游戏——————-\n”);
}

void setmine(char board[ROWS][COLS], int row, int col) {
int count = MINE_COUNT;
while (count) {
int x = rand() % row + 1;
int y = rand() % col + 1;
if (board[x][y] == ‘0’) {
board[x][y] = ‘1’;
count–;
}
}
}

int get_mine_count(char board[ROWS][COLS], int x, int y) {
return (board[x - 1][y] +
board[x - 1][y - 1] +
board[x - 1][y + 1] +
board[x][y - 1] +
board[x][y + 1] +
board[x + 1][y - 1] +
board[x + 1][y + 1] +
board[x + 1][y] - 8 * ‘0’);
}

// 扩展空白区域的递归函数
void expand(char mine[ROWS][COLS], char show[ROWS][COLS], int x, int y, int* win) {
// 检查坐标是否越界
if (x < 1 || x > ROW || y < 1 || y > COL) {
return;
}

// 如果该位置已经展开或者是地雷，则返回
if (show[x][y] != '*' || mine[x][y] == '1') {
	return;
}

// 计算周围地雷数量
int count = get_mine_count(mine, x, y);
show[x][y] = count + '0';
(*win)++;

// 如果周围没有地雷，则递归展开周围的8个格子
if (count == 0) {
	expand(mine, show, x - 1, y - 1, win);
	expand(mine, show, x - 1, y, win);
	expand(mine, show, x - 1, y + 1, win);
	expand(mine, show, x, y - 1, win);
	expand(mine, show, x, y + 1, win);
	expand(mine, show, x + 1, y - 1, win);
	expand(mine, show, x + 1, y, win);
	expand(mine, show, x + 1, y + 1, win);
}

}

void FindMine(char mine[ROWS][COLS], char show[ROWS][COLS], int row, int col) {
int x = 0;
int y = 0;
int win = 0;

while (win < row * col - MINE_COUNT) {
	printf("牢底祝你这次不会被炸死，坐标>");
	scanf_s("%d %d", &x, &y);
	if (x >= 1 && x <= row && y >= 1 && y <= col) {
		if (show[x][y] != '*') {
			printf("会玩不这已经有了\n");
		}
		else {
			if (mine[x][y] == '1') {
				printf("很遗憾，您踩到地雷了！游戏结束\n");
				paly_board(mine, ROW, COL);
				break;
			}
			else {
				// 使用扩展函数而不是直接显示数字
				expand(mine, show, x, y, &win);
				paly_board(show, ROW, COL);
			}
		}
	}
	else {
		printf("牢底会不会玩，重新输入！\n");
	}

	if (win == row * col - MINE_COUNT) {
		printf("牢底你会玩，赢了\n");
		paly_board(show, ROW, COL);
	}
}

}

game.h:
#pragma once

#include <stdio.h>
#include <stdlib.h>
#include <time.h>

#define ROW 9
#define COL 9

#define ROWS ROW+2
#define COLS COL+2

#define MINE_COUNT 10

// 初始化棋盘
void in_board(char board[ROWS][COLS], int rows, int cols, char set);

// 打印棋盘
void paly_board(char board[ROWS][COLS], int row, int col);

// 布置雷
void setmine(char board[ROWS][COLS], int row, int col);

// 排雷
void FindMine(char mine[ROWS][COLS], char show[ROWS][COLS], int row, int col);

// 扩展空白区域 - 新增函数
void expand(char mine[ROWS][COLS], char show[ROWS][COLS], int x, int y, int* win);

test.c:
#include “game.h”

void menu() {
printf(“\n”);
printf(“* 1. 牢底来扫雷了 \n”);
printf(“ 2. 不想玩就输这个 *******\n”);
printf(“******\n”);
}

void game() {
char mine[ROWS][COLS] = {0};//存放布置好的雷
char show[ROWS][COLS] = {0};//存放排查出的雷的信息
//初始化棋盘
in_board(mine, ROWS, COLS, ‘0’);//‘0’
in_board(show, ROWS, COLS, ‘‘);//‘‘
//布置雷
setmine(mine, ROW, COL);
//打印棋盘
//paly_board(mine, ROW, COL);
paly_board(show, ROW, COL);
//排查雷
FindMine(mine, show, ROW, COL);
}

int main() {
int input = 0;
srand((unsigned int)time(NULL));
do {
menu();
printf(“请选择:>”);
scanf_s(“%d”, &input);
switch (input) {
case 1:
game();
break;
case 2:
printf(“退出游戏\n”);
break;
default:
printf(“选择错误，重新选择！\n”);
break;
}
} while (input);
return 0;
}
