while (1) {

int ping(char board[ROW][COL], int row, int col) {
int i = 0;
int j = 0;
for (i = 0; i < col; i++) {
for (j = 0; j < col; j++) {
if (board[i][j] == ' ') {
return 0;
}
}
}
return 1;
}
char win(char board[ROW][COL], int row, int col) {
int i = 0; //行
for (i = 0; i < row; i++) {
if (board[i][0] == board[i][1] && board[i][2] && board[i][1] != ' ') {
return board[i][1];
}
}
int j = 0; //列
for (j = 0; j < row; j++) {
if (board[0][j] == board[1][j] && board[2][j] && board[1][j] != ' ') {
return board[1][j];
}
}
if(board[0][2] == board[1][1] && board[2][0] == board[1][1] && board[1][1] != ' '){
return board[1][1]; } //对角线

if (ping(board, row, col)) {
return'Q';
}
return'C';
}

#include <stdio.h>
#include "game.h"
void menu() {
printf("########################################\n");
printf("################1.开始玩耍###############\n");
printf("################2.离开游戏###############\n");
} //设计个菜单
void game() {
char ret = 0; //加个变量接收值
char board[ROW][COL] = { 0 };
in_board(board, ROW, COL); //初始化棋盘
out_board(board, ROW, COL); //耍时候打印的棋盘
while (1) { //下棋
people(board, ROW, COL);
ret = win(board, ROW, COL); //判断输赢
if (ret != 'C') {
break;
}
out_board(board, ROW, COL);
computer(board, ROW, COL);
ret = win(board, ROW, COL);
if (ret != 'C') {
break;
}
out_board(board, ROW, COL);
}
if (ret == '*') {
printf("牢底666赢了\n");
}
else if (ret == '#') {
printf("牢底你没下过电脑\n");
}
else {
printf("这都能平局，菜！");
}
out_board(board, ROW, COL);
}
int main() {
srand((unsigned int)time(NULL)); //利用时间栈制作随机数
int input = 0;
do {
menu(); //设置循环，方便玩耍
printf("你想选啥？>");
scanf_s("%d", &input);
switch (input) {
case 1:
printf("下棋了牢弟\n");
game(); //设计游戏
break;
case 2:
printf("想退选这");
break;
default:
printf("选个数字都不会，菜！");
break;
}
} while (input);
return 0;
}
