include <stdio.h>
#include <stdbool.h>

int main() {
int n, m, horse_x, horse_y;
scanf(“%d %d %d %d”, &n, &m, &horse_x, &horse_y);

// 马的控制点偏移量
int dx[] = {1, 1, -1, -1, 2, 2, -2, -2, 0};
int dy[] = {2, -2, 2, -2, 1, -1, 1, -1, 0};

// 创建障碍物网格
bool obstacle[n + 1][m + 1];
for (int i = 0; i <= n; i++) {
    for (int j = 0; j <= m; j++) {
        obstacle[i][j] = false;
    }
}

// 标记马的控制点为障碍物
for (int i = 0; i < 9; i++) {
    int nx = horse_x + dx[i];
    int ny = horse_y + dy[i];
    if (nx >= 0 && nx <= n && ny >= 0 && ny <= m) {
        obstacle[nx][ny] = true;
    }
}

// 如果起点是障碍物，直接输出0
if (obstacle[0][0]) {
    printf("0\n");
    return 0;
}

// 初始化DP数组
long long dp[n + 1][m + 1];
for (int i = 0; i <= n; i++) {
    for (int j = 0; j <= m; j++) {
        dp[i][j] = 0;
    }
}
dp[0][0] = 1;

// 动态规划计算路径数
for (int i = 0; i <= n; i++) {
    for (int j = 0; j <= m; j++) {
        if (obstacle[i][j]) {
            continue;
        }
        if (i > 0 && !obstacle[i - 1][j]) {
            dp[i][j] += dp[i - 1][j];
        }
        if (j > 0 && !obstacle[i][j - 1]) {
            dp[i][j] += dp[i][j - 1];
        }
    }
}

printf("%lld\n", dp[n][m]);
return 0;

}
