计算奇数和偶数个数
#include<stdio.h>
int main(){
int n ;
int count1 = 0;
int count2 = 0;
int arr[1000];
scanf(“%d”,&n);
for(int i = 0;i<n;i++){
scanf(“%d”,&arr[i]);
if(arr[i]%2==1){
count1++;
}
else{
count2++;
}
}
printf(“%d %d”,count1,count2);
return 0;

}
