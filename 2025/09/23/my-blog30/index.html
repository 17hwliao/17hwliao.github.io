using namespace std;

const double q = 1e-10;

bool func1(vector& arr) {
if (arr.size() == 1) {
return fabs(arr[0] - 24) < q;
}

int n = arr.size();
for (int i = 0; i < n; i++) {
    for (int j = i + 1; j < n; j++) {
        double a = arr[i], b = arr[j];
        vector<double> next;
        for (int k = 0; k < n; k++) {
            if (k != i && k != j) {
                next.push_back(arr[k]);
            }
        }
        vector<double> brr = {a + b, a - b, b - a, a * b};
        if (fabs(b) > q) brr.push_back(a / b);
        if (fabs(a) > q) brr.push_back(b / a);
        for (const auto& v : brr) {
            next.push_back(v);
            if (func1(next)) {
                return true;
            }
            next.pop_back();
        }
    }
}
return false;

}

bool func(vector& nums) {
vector arr(nums.begin(), nums.end());
return func1(arr);
}

int main() {
int n;
cin >> n;
bool first = true;

while (n > 0) {
    vector<int> nums(4);
    for (int i = 0; i < 4; i++) {
        cin >> nums[i];
    }
    
    if (!first) cout << " ";
    first = false;
    if (func(nums)) {
        cout << "True";
    } else {
        cout << "False";
    }
    n--;
}

return 0;

}
