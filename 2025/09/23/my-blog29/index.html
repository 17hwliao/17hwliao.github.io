#include<bits/stdc++.h>
using namespace std;

void func(vector arr, int n, int m) {
for(int i = 0; i < n; i += 2) {
cout << arr[i] << ‘\n’;
}
int start = n - 1;
if (n % 2 == 1) {
start = n - 2;
}

for(int i = start; i >= 0; i -= 2) {
    cout << arr[i] << '\n';
}

}

int main() {
int n = 1;
int count = 1;

while (cin >> n && n != 0) {
    vector<string> arr;
    
    for(int i = 0; i < n; i++) {
        string str;
        cin >> str;
        arr.push_back(str);
    }
    cout << "SET " << count << '\n';
    func(arr, n, 0);
    count++;
}
return 0;

}
