title: my-blog28
date: 2025-09-23 22:06:00

#include <bits/stdc++.h>
using namespace std;
void hanoi(int n, char a, char b, char c) {
if (n == 1) {
cout << a << “–>” << b << endl;
return;
}
hanoi(n - 1, a, c, b);
cout << a << “–>” << b << endl;
hanoi(n - 1, c, b, a);
}

int main() {
int n;
cin >> n;
hanoi(n, ‘a’, ‘b’, ‘c’);
return 0;
}
