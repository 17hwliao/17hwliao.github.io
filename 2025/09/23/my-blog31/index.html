补档九月二十

title: my-blog31
date: 2025-09-23 22:24:28
tags:

#include <bits/stdc++.h>
using namespace std;
typedef vector<vector> a;
int main() {
a arr(4, vector(5));
for(int i = 0; i < 4; i++) {
for(int j = 0; j < 5; j++) {
cin >> arr[i][j];
}
}
for(int j = 0; j < 5; j++) {
for(int k = 0; k < 3; k++) {
for(int i = 0; i < 3 - k; i++) {
if(arr[i][j] > arr[i+1][j]) {
long long tempt = arr[i][j];
arr[i][j] = arr[i+1][j];
arr[i+1][j] = tempt;
}
}
}
}
for(int i = 0; i < 4; i++) {
for(int j = 0; j < 5; j++) {
cout << setw(4) << arr[i][j];
}
cout << endl;
}
return 0;
}
