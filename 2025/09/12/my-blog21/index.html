#include<bits/stdc++.h>
typedef long long ll;

using namespace std;

int main(){
ll n;
cin>>n;
for(int i = 0; i < n; i++) {
char str[5];
scanf(“%s”,str);
ll sec = 0;
int current = 0;
int count = 0; 

    for(char c : str) {  
        if(count >= 4) break;  
        int target = c - '0';
        sec += abs(target - current); 
        sec++; 
        current = target;
        count++;  
    }
    printf("%lld\n",sec);
}
return 0;

}
