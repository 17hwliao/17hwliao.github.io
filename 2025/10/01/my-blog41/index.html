#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int main(){
ll n, m, k;
cin >> n >> m >> k;
vector<set> arr(k + 1);
for(int i = 0; i < n; i++){
for(int j = 0; j < m; j++){
int day;
cin >> day;
arr[day].insert(j + 1);
}
}
for(int day = 1; day <= k; day++){
cout << arr[day].size();
if(day < k) {
cout << “ “;
}
}
cout << endl;

return 0;

}
