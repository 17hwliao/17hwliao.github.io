#include<bits/stdc++.h>
using namespace std;
int main(){
string str1,str2;
getline(cin,str1);
getline(cin,str2);
for(auto& c : str1 ){
c = tolower(static_cast (c));
}
for(auto& m: str2 ){
m = tolower(static_cast (m));
}
int count = -1;
int n = 0;
int j = 0 ;
int y = -1;
int b = 0;
for(int i = 0; i <= (int)(str2.size() - str1.size()); i++) {
if(i == 0 || str2[i-1] == ‘ ‘) {
bool match = true;
for(n = 0; n < (int)str1.size(); n++) {
if(str2[i + n] != str1[n]) {
match = false;
break;
}
}
if(match && (i + str1.size() == str2.size() || str2[i + str1.size()] == ‘ ‘)) {
b++;
if(b == 1) {
count = i;
}
}
}
}

if(b == 0){
    cout<<-1;
}else{
    cout<<b<<' '<<count;
}
return 0;

}
