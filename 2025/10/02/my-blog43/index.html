#include <bits/stdc++.h>
using namespace std;
int main() {
ios::sync_with_stdio(false);
cin.tie(nullptr);
int n;
cin >> n;
vector even(n), odd(n);
vector len(n);
for (int i = 0; i < n; i++) {
string s;
cin >> s;
len[i] = s.size();
for (int j = 0; j < s.size(); j++) {
int digit = s[j] - ‘0’;
if (j % 2 == 0) even[i] += digit;
else odd[i] += digit;
}
}
long long count = 0;
for (int i = 0; i < n; i++) {
for (int j = 0; j < n; j++) {
if (i == j) continue;

        long long total_even, total_odd;
        if (len[i] % 2 == 0) {
            total_even = even[i] + even[j];
            total_odd  = odd[i]  + odd[j];
        } else {
            total_even = even[i] + odd[j];
            total_odd  = odd[i]  + even[j];
        }

        long long diff = total_odd - total_even;
        if (diff % 11 == 0) {
            count++;
        }
    }
}
cout << count << '\n';
return 0;

}
