##算法一:排序
//实现快排的算法
void quick_sort(int arr[],int l ,int r) {
if (l >= r) return; //终止条件不要忘记
int x = arr[l];
int i = l - 1; int j = r + 1;
while (i < j) { //控制指针移动
do i++; while (arr[i] < x);
do j–; while (arr[j] > x);
if (i < j)swap(arr[i], arr[j]);
}
quick_sort(arr, l, j);
quick_sort(arr, j+1, r);
}(在调用函数时传入r =n-1,控制全部范围)
(部分时候调用中间值(即x = arr[(l+r)>>1])快排会优化最坏(O(nlog(n))))
//通过快速排序找数字(注意下返回的是数组的元素int)
int quick_sort(int arr[],int k ,int l , int r){
if(l>=r) return arr[l];
int i = l-1;int j = r+1;
int x = arr[(l+r)/2];
while(i<j){
do i++;while(arr[i]<x);
do j–;while(arr[j]>x);
if(i<j) swap(arr[i],arr[j]);
}
int pos = j - l + 1;
if(pos>=k) return quick_sort(arr,k,l,j);
return quick_sort(arr,k - pos,j+1,r);
}

void merge_sort(int arr[], int l, int r) {
if (l >= r) return;
int mid = (l + r) >> 1;
merge_sort(arr, l, mid); merge_sort(arr, mid + 1, r);
