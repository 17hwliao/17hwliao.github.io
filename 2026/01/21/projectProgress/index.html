#include<graphics.h>
#include<malloc.h>
#include<string.h>
#include<mmsystem.h>
#pragma comment(lib,"winmm.lib") //链接器所需要的库文件
//封装按钮
struct Button {
int x;
int y;
int w;
int h;
unsigned long curColor;
unsigned long oldColor; //后期变色使用
char* str;
};
//创建按钮
struct Button* creatButton(int x, int y, int w, int h, unsigned long curColor, const char* str) {
struct Button* pButton = (struct Button*)malloc(sizeof(struct Button));
if (pButton == NULL) return NULL;
pButton->x = x;
pButton->y = y;
pButton->w = w;
pButton->h = h;
pButton->curColor = curColor;
pButton->oldColor = curColor;
int length = strlen(str);
pButton->str = (char*)malloc(sizeof(char) * (length + 1));
if (pButton->str == NULL) return NULL;
strcpy_s(pButton->str, length + 1, str); //安全的字符串拷贝,检查大小,阻止溢出
return pButton;
};
//显示按钮
void Show(struct Button* pButton) {
//设置颜色
setfillcolor(pButton->curColor);
//显示矩形
solidrectangle(pButton->x, pButton->y, pButton->x + pButton->w, pButton->y + pButton->h);
//设置本地字体
settextstyle(18, 0, "STXINGKA");
//居中处理
int textw = textwidth(pButton->str);
int texth = textheight(pButton->str);
int xx = pButton->x + (pButton->w - textw) / 2;
int yy = pButton->y + (pButton->h - texth) / 2;
//显示文字
setbkmode(TRANSPARENT); //去除文字背景
settextcolor(BLACK); //设置文字颜色
outtextxy(xx, yy, pButton->str);
}
//设置按钮状态
int InButton(struct Button* pButton, ExMessage msg) {
//判断鼠标是否在按钮区域内
if (msg.x >= pButton->x && msg.x <= pButton->x + pButton->w && msg.y >= pButton->y && msg.y <= pButton->y + pButton->h) {
pButton->curColor = LIGHTBLUE;
return 1;
}
pButton->curColor = pButton->oldColor;
return 0;
}
//设置按钮文本
void SetTextButton(struct Button* pButton, const char* str) {
int length = strlen(str);
free(pButton->str);
//再次内存申请 保证安全
pButton->str = (char*)malloc(sizeof(str) * (length + 1));
if (pButton->str == NULL) return;
strcpy_s(pButton->str, length + 1, str);
}

int main() {

//调用initgraphy函数创建窗口 (宽度 , 高度)
initgraph(800, 600);
//line(0,0,800,600) ; 
IMAGE img;  //图片类型 

loadimage(&img, "./source/picture/background.jpg", 800, 600);

//创建按钮
struct Button* play = creatButton(0, 0, 100, 25, RGB(255, 255, 148), "Man!");
struct Button* pause = creatButton(0, 30, 100, 25, RGB(255, 255, 148), "Manba Wait");
//struct Button* resume = creatButton(0, 60, 100, 25, RGB(255, 255, 148), "Manba come");
struct Button* close = creatButton(0, 60, 100, 25, RGB(255, 255, 148), "Manba out");

mciSendString("open ./source/music/为时已晚.mp3 alias mymusic", 0, 0, 0); //控制播放音乐

BeginBatchDraw();
ExMessage msg;
static int flag = 0;
while (1) {
	putimage(0, 0, &img);
	Show(play);
	Show(pause);
	Show(close);

	FlushBatchDraw();

	while (peekmessage(&msg)) {    //判断新鼠标操作再处理 (若不循环判断处理 , 会反应多次)
		if (InButton(play, msg) && msg.message == WM_LBUTTONDOWN) {
			mciSendString("play mymusic", 0, 0, 0);
		}if (InButton(pause, msg) && msg.message == WM_LBUTTONDOWN) {
			if (flag == 0) {
				SetTextButton(pause, "暂停");
				mciSendString("pause mymusic", 0, 0, 0);
				flag = 1;
			}
			else if (flag == 1) {
				SetTextButton(pause, "继续");
				mciSendString("resume mymusic", 0, 0, 0);
				flag = 0;
			}

		}if (InButton(close, msg) && msg.message == WM_LBUTTONDOWN) {
			mciSendString("close mymusic", 0, 0, 0);
		}
	}

}

EndBatchDraw();
//关闭图行窗口
closegraph();
return 0;

}
