题目1 : bfs 找宝藏 : 设计动态数组 , 大小resize 设置题目大小
typedef pair<int, int > PII;
vector g; //创建储存数组
vector<vector> visited; //标记状态

bool bfs(int x, int y, int n, int m) {
初始化 : 队列传值, 初始状态 ;
queue q;
q.push(make_pair(x, y));
visited[x][y] = true;

bool tre = false; 					//判断有无宝藏
int dx[4] = { 0, 1, -1, 0 };
int dy[4] = { 1, 0, 0, -1 };

//开始对队列循环
while (!q.empty()) {
//判断队列中元素是否达标:
int qx = q.front().first;
int qy = q.front().second;
q.pop();
if (g[qx][qy] >= '2' && g[qx][qy] <= '9') {
tre = true;
}
//遍历可能的全部方向 ;
for (int d = 0; d < 4; d++) {
int nx = qx + dx[d];
int ny = qy + dy[d];
//判断条件 + 压入队列 + 标记 ;
if (nx >= 0 && nx < n && ny >= 0 && ny < m
&& g[nx][ny] != '0' && !visited[nx][ny]) {
visited[nx][ny] = true;
q.push(make_pair(nx, ny));
}
}
}
按照思路返回值 ;
return tre;
}

int main() {
int n, m;
cin >> n >> m;

g.resize(n);
visited.resize(n, vector<bool>(m, false));

for (int i = 0; i < n; i++) {
    cin >> g[i];
}

int island = 0;
int tre = 0;

for (int i = 0; i < n; i++) {
    for (int j = 0; j < m; j++) {
        if (g[i][j] != '0' && !visited[i][j]) {
            island++;
            if (bfs(i, j, n, m)) {
                tre++;
            }
        }
    }
}

cout << island << " " << tre << endl;

return 0;

}

题目2 : dijkstra算法找最短路径 ;
typedef pair<int, int> PII;
const int N = 2e5 + 17 ;
//按照数据稀疏使用邻接表 , 邻接矩形
int h[N], e[N * 2], ne[N * 2], idx;
int w[N * 2]; //存储权值
int dist[N]; //路径长度
bool visited[N];

void add(int a, int b, int c) {
e[idx] = b;
ne[idx] = h[a];
w[idx] = c;
h[a] = idx++;
}

int dijkstra(int n) {
memset(dist, 0x3f, sizeof dist);
dist[1] = 0;

priority_queue<PII, vector<PII>, greater<PII>> heap;   //小顶堆 ; <长度 , 序号> , 自动排序, 实现取出距离最小的数
heap.push({0, 1});

while (!heap.empty()) {
    auto t = heap.top();
    heap.pop();
    //防止相同序号再次进入堆
    if (visited[t.second]) continue;
    visited[t.second] = true;

//找寻最短路径 , h[ 序号 ] 进入链表编号,
for (int i = h[t.second]; i != -1; i = ne[i]) {
int v = e[i]; // v 表示下一节点 ;
//更新最短路径
if (dist[v] > t.first + w[i]) {
dist[v] = t.first + w[i];
heap.push({dist[v], v});
}
}
}
return dist[n];
}

signed main() {
ios::sync_with_stdio(false);
cin.tie(nullptr);

int N;
cin >> N;

memset(h, -1, sizeof h);
idx = 0;

for (int i = 1; i < N; i++) {
    int A, B, X; 
    cin >> A >> B >> X;
	通过 加A 可以去 i+1 ; 
	通过 加B 可以去 X ; 
    add(i, i + 1, A);
    add(i, X, B);
}

// 调用 Dijkstra
int result = dijkstra(N);

cout << result << endl;

return 0;

}

信息传递 并查集算法:
const int MAXN = 200005;
int parent[MAXN], dist[MAXN];

int find(int x) {
if (parent[x] != x) {
int root = find(parent[x]); //储存根节点 ;
dist[x] += dist[parent[x]]; //记录到祖节点的距离;
parent[x] = root; //并查集 ;
}
return parent[x];
}

int main() {
int n;
cin >> n;

for (int i = 1; i <= n; i++) {
    parent[i] = i;
    dist[i] = 0;
}

int ans = n + 1;
for (int i = 1; i <= n; i++) {
    int t;
    cin >> t;
    
    find(i);
    find(t);
    
    int root_i = find(i);
    int root_t = find(t);
    
    if (root_i != root_t) {       //判断是否同主族 ; 
        parent[root_i] = root_t;    //并集
        dist[root_i] = 1 + dist[t] - dist[i];    //储存路径长度 ; 
    }
    else {
        ans = min(ans, abs(dist[i] - dist[t]) + 1);     // 找出最短路径 ; 
    }
}

cout << ans << endl;
return 0;

}

使用拓扑算法实现找最小环 :
const int N = 200010;

int n;
int h[N], e[N], ne[N], idx; // 邻接表
int d[N]; // 入度
int p[N]; // 拓扑序列（本题不需要输出）
bool st[N]; // 访问标记

void add(int a, int b) {
e[idx] = b;
ne[idx] = h[a];
h[a] = idx++;
}

int main() {
ios::sync_with_stdio(false);
cin.tie(nullptr);

cin >> n;

// 初始化
memset(h, -1, sizeof h);
idx = 0;

// 读取数据
for (int i = 1; i <= n; i++) {
    int t;
    cin >> t;
    add(i, t);  // i -> t
    d[t]++;     // 入度+1
}

// 拓扑排序
queue<int> q;
for (int i = 1; i <= n; i++) {
    if (!d[i]) {
        q.push(i);
    }
}

while (!q.empty()) {
    int t = q.front();
    q.pop();
    
    for (int i = h[t]; i != -1; i = ne[i]) {
        int j = e[i];
        d[j]--;
        if (!d[j]) {
            q.push(j);
        }
    }
}

// 找最小环
int ans = n + 1;

for (int i = 1; i <= n; i++) {
    if (d[i] > 0 && !st[i]) {
        int len = 0;
        int u = i;
        
        while (!st[u]) {
            st[u] = true;
            // 注意：每个人只有一个出边，所以h[u]就是第一条边
            u = e[h[u]];
            len++;
        }
        
        ans = min(ans, len);
    }
}

cout << ans << endl;

return 0;

}
